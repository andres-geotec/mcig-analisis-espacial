movilidad <- read.csv(file.path(ruta_data, "municipio-interno.csv")) |>
  mutate(# asegurar los caracteres adecuados con ceros a la izquierda
    source = str_pad(as.character(source), width = 5, pad = "0", side = "left"),
    target = str_pad(as.character(target), width = 5, pad = "0", side = "left"))

orden <- sort(unique(movilidad$source))

# Convertir a factor con niveles ordenados
matriz_od <- movilidad |>
  mutate(source = factor(source, levels = orden),
         target = factor(target, levels = orden)) |>
  group_by(source, target, .drop = FALSE) |>
  summarise(w = sum(w), .groups = 'drop') |>
  pivot_wider(names_from = target, values_from = w, values_fill = 0)
